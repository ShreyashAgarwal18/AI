// Assignment 5 - Artificial Intelligence
// Single-Source Shortest Path Problem - Dijkstra's Algorithm
//
// Given a weighted graph and a source vertex, find the shortest
// distance from the source to every other vertex.
//
// Works for graphs with NON-NEGATIVE edge weights only.

#include <iostream>
#include <climits>
using namespace std;

const int V = 6;            // number of vertices

// ============================================================
// Find the unvisited vertex with the smallest known distance
// ============================================================
int minDistance(int dist[], bool visited[]) {
    int min = INT_MAX, minIndex = -1;
    for (int v = 0; v < V; v++) {
        if (!visited[v] && dist[v] < min) {
            min = dist[v];
            minIndex = v;
        }
    }
    return minIndex;
}

// ============================================================
// Dijkstra's Algorithm
// graph[u][v] = weight of edge u-v, 0 means no edge
// ============================================================
void dijkstra(int graph[V][V], int src) {
    int dist[V];               // shortest distance from src to each vertex
    bool visited[V];           // visited[i] = true if shortest dist found

    // Step 1: Initialize all distances as INF and visited as false
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        visited[i] = false;
    }
    dist[src] = 0;             // distance to source is 0

    // Step 2: Find shortest path for all V vertices
    for (int count = 0; count < V - 1; count++) {

        // Pick the unvisited vertex with the smallest distance
        int u = minDistance(dist, visited);
        if (u == -1) break;
        visited[u] = true;

        // Update distances of all neighbours of u
        for (int v = 0; v < V; v++) {
            // If there's an edge u-v and v is unvisited
            // and going through u gives a shorter path
            if (!visited[v] && graph[u][v] != 0
                && dist[u] != INT_MAX
                && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }

    // Step 3: Print results
    cout << "\nShortest distances from vertex " << src << ":\n";
    cout << "Vertex\tDistance\n";
    for (int i = 0; i < V; i++) {
        cout << "  " << i << "\t   " << dist[i] << endl;
    }
}

// ============================================================
// MAIN
// ============================================================
int main() {
    // Weighted undirected graph (adjacency matrix)
    // 0 means no direct edge
    int graph[V][V] = {
        {0, 4, 0, 0, 0, 0},
        {4, 0, 8, 0, 0, 0},
        {0, 8, 0, 7, 0, 4},
        {0, 0, 7, 0, 9, 14},
        {0, 0, 0, 9, 0, 10},
        {0, 0, 4, 14, 10, 0}
    };

    cout << "=== Dijkstra's Single-Source Shortest Path ===\n\n";
    cout << "Graph (adjacency matrix):\n";
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) cout << graph[i][j] << "\t";
        cout << endl;
    }

    dijkstra(graph, 0);
    return 0;
}
