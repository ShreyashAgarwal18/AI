// Assignment 6 - Artificial Intelligence
// Prim's Minimum Spanning Tree (MST) Algorithm
//
// Builds a tree connecting all vertices with the minimum total edge weight.
// Greedy approach: start from any vertex, then keep adding the
// smallest edge that connects a NEW vertex to the tree.

#include <iostream>
#include <climits>
using namespace std;

const int V = 5;            // number of vertices

// ============================================================
// Find the unvisited vertex with the smallest key value
// ============================================================
int minKey(int key[], bool inMST[]) {
    int min = INT_MAX, minIndex = -1;
    for (int v = 0; v < V; v++) {
        if (!inMST[v] && key[v] < min) {
            min = key[v];
            minIndex = v;
        }
    }
    return minIndex;
}

// ============================================================
// Prim's Algorithm
// graph[u][v] = edge weight (0 = no edge)
// ============================================================
void primMST(int graph[V][V]) {
    int parent[V];           // parent[i] = parent of vertex i in the MST
    int key[V];              // key[i] = smallest edge weight connecting i to MST
    bool inMST[V];           // true if vertex is already included in MST

    // Step 1: Initialize all keys as INF, no vertex in MST yet
    for (int i = 0; i < V; i++) {
        key[i] = INT_MAX;
        inMST[i] = false;
    }

    // Start with vertex 0 (key=0 makes it picked first)
    key[0] = 0;
    parent[0] = -1;          // root of MST has no parent

    // Step 2: Build MST with V-1 edges
    for (int count = 0; count < V - 1; count++) {

        // Pick the smallest-key vertex not yet in MST
        int u = minKey(key, inMST);
        inMST[u] = true;

        // Update keys of neighbours of u
        for (int v = 0; v < V; v++) {
            // If edge u-v exists, v is unvisited,
            // and edge weight is smaller than current key[v]
            if (graph[u][v] != 0 && !inMST[v] && graph[u][v] < key[v]) {
                key[v] = graph[u][v];
                parent[v] = u;
            }
        }
    }

    // Step 3: Print the MST edges
    int totalWeight = 0;
    cout << "\nMinimum Spanning Tree edges:\n";
    cout << "Edge\tWeight\n";
    for (int i = 1; i < V; i++) {
        cout << " " << parent[i] << " - " << i << "\t  " << graph[i][parent[i]] << endl;
        totalWeight += graph[i][parent[i]];
    }
    cout << "\nTotal weight of MST = " << totalWeight << endl;
}

// ============================================================
// MAIN
// ============================================================
int main() {
    // Weighted undirected graph
    int graph[V][V] = {
        {0, 2, 0, 6, 0},
        {2, 0, 3, 8, 5},
        {0, 3, 0, 0, 7},
        {6, 8, 0, 0, 9},
        {0, 5, 7, 9, 0}
    };

    cout << "=== Prim's Minimum Spanning Tree ===\n\n";
    cout << "Graph (adjacency matrix):\n";
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) cout << graph[i][j] << "\t";
        cout << endl;
    }

    primMST(graph);
    return 0;
}
