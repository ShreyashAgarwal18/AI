#include <iostream>
using namespace std;

#define V 5

int main() {

    int graph[V][V] = {
        {0, 9, 75, 0, 0},
        {9, 0, 95, 19, 42},
        {75, 95, 0, 51, 66},
        {0, 19, 51, 0, 31},
        {0, 42, 66, 31, 0}
    };

    bool selected[V] = {false};

    int edges = 0;
    int totalCost = 0;

    // Start from vertex 0
    selected[0] = true;

    cout << "Edges in Minimum Spanning Tree:\n";

    while (edges < V - 1) {

        int min = 9999;
        int x = 0;
        int y = 0;

        for (int i = 0; i < V; i++) {

            if (selected[i]) {

                for (int j = 0; j < V; j++) {

                    if (!selected[j] && graph[i][j]) {

                        if (min > graph[i][j]) {
                            min = graph[i][j];
                            x = i;
                            y = j;
                        }
                    }
                }
            }
        }

        cout << x << " - " << y << " : " << graph[x][y] << endl;

        totalCost = totalCost + graph[x][y];

        selected[y] = true;

        edges++;
    }

    cout << "\nTotal Cost = " << totalCost << endl;

    return 0;
}
