// Assignment 1 - Artificial Intelligence
// Depth First Search (DFS - recursive) and
// Breadth First Search (BFS - iterative using queue)
// on an undirected graph.

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

// ============================================================
// Graph stored as an adjacency list
// adj[u] = list of vertices adjacent to vertex u
// ============================================================
const int N = 7;                        // number of vertices (we use 1..6)
vector<int> adj[N];

// Add an undirected edge between u and v
void addEdge(int u, int v) {
    adj[u].push_back(v);
    adj[v].push_back(u);
}

// ============================================================
// DFS - Depth First Search (RECURSIVE)
// Visits a node, marks it, then recursively visits each
// unvisited neighbour.
// ============================================================
void dfs(int node, bool visited[]) {
    visited[node] = true;
    cout << node << " ";
    for (int neighbour : adj[node]) {
        if (!visited[neighbour])
            dfs(neighbour, visited);    // recursive call
    }
}

// ============================================================
// BFS - Breadth First Search (ITERATIVE using a queue)
// Visits nodes level by level (closest first).
// ============================================================
void bfs(int start) {
    bool visited[N] = {false};
    queue<int> q;
    q.push(start);
    visited[start] = true;

    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";

        for (int neighbour : adj[node]) {
            if (!visited[neighbour]) {
                visited[neighbour] = true;
                q.push(neighbour);
            }
        }
    }
}

// ============================================================
// MAIN
// ============================================================
int main() {
    // Build the graph (undirected)
    //         1
    //        / \
    //       2   3
    //      / \   \
    //     4   5   6
    //          \ /
    //           (5-6)
    addEdge(1, 2);
    addEdge(1, 3);
    addEdge(2, 4);
    addEdge(2, 5);
    addEdge(3, 6);
    addEdge(5, 6);

    cout << "Graph adjacency list:\n";
    for (int i = 1; i <= 6; i++) {
        cout << "  " << i << " -> ";
        for (int v : adj[i]) cout << v << " ";
        cout << endl;
    }

    int start = 1;

    cout << "\nDFS starting from " << start << ": ";
    bool visited[N] = {false};
    dfs(start, visited);

    cout << "\nBFS starting from " << start << ": ";
    bfs(start);
    cout << endl;

    return 0;
}
